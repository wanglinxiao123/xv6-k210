# 结构体
```C
struct run
{
    struct run *next;
};

struct
{
    struct spinlock lock;
    struct run *freelist; // 内存链表头
    uint64 npage;         // 空余的页面数
} kmem;
```
# 函数
```C
// 初始化自旋锁，将空余空间回收到链表
void kinit()

// 回收 (pa_start, pa_end) 对应的物理页面
void freerange(void *pa_start, void *pa_end)

// 回收 pa 对应的物理页面
void kfree(void *pa)

// 从空闲链表头中获取 4KB，返回物理地址
void *kalloc(void)

// 统计空闲物理内存总数
uint64 freemem_amount(void)
```
